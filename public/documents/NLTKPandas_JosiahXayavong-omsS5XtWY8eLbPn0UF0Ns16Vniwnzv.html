punctuation = ["\"", ",", ".", "-", "?", "'", "(", ")", "--", ":", ";", "/", "\\"]
basic_stopwords = ['and', 'or', 'the', 'a', 'an', 'to', 'with', 'in', 'on', 'for', 'of', 'at', 'by', 'from']

# Define the cleaning function
def clean_skills(text):
    text = text.lower()
    for p in punctuation:
        text = text.replace(p, "")
    words = text.split()
    words = [w for w in words if w not in basic_stopwords]
    return ' '.join(words)

# Apply the function to create the new column
df['t2_sk'] = df['Applicant Skills'].apply(clean_skills)

# Display the result
df[['Applicant Skills', 't2_sk']]
